model User {
  id           String       @id @default(uuid()) @db.Uuid
  username     String       @unique @db.VarChar(50)
  email        String       @unique @db.VarChar(100)
  passwordHash String       @map("password_hash") @db.VarChar(255)
  fullName     String       @map("full_name") @db.VarChar(100)
  phoneNumber  String?      @map("phone_number") @db.VarChar(30)
  avatarUrl    String?      @map("avatar_url")
  bio          String?
  city         String?      @db.VarChar(100)
  gender       UserGender?
  dateOfBirth  DateTime?    @map("date_of_birth") @db.Date
  role         UserRole     @default(PLAYER)
  isVerified   Boolean      @default(false) @map("is_verified")
  isActive     Boolean      @default(true) @map("is_active")
  lastLoginAt  DateTime?    @map("last_login_at") @db.Timestamptz(6)

  createdAt    DateTime     @default(now()) @map("created_at") @db.Timestamptz(6)
  updatedAt    DateTime     @updatedAt @map("updated_at") @db.Timestamptz(6)
  deletedAt    DateTime?    @map("deleted_at") @db.Timestamptz(6)

  // auth
  sessions UserSession[]

  // preferences
  favoriteSports UserFavoriteSport[]

  // venues
  ownedVenues    Venue[] @relation("VenueOwner")
  approvedVenues Venue[] @relation("VenueApprover")

  // bookings
  bookings          Booking[] @relation("BookingUser")
  cancelledBookings Booking[] @relation("BookingCancelledBy")

  // payments
  reviewedPayments    BookingPayment[]   @relation("PaymentReviewer")
  paymentReviewLogs   PaymentReviewLog[] @relation("PaymentReviewActor")

  // community
  posts        Post[]
  postLikes    PostLike[]
  postComments PostComment[]

  // matches
  hostedMatches Match[]            @relation("MatchHost")
  joinedMatches MatchParticipant[]

  // reviews
  venueReviews VenueReview[]

  // notifications
  notifications Notification[]

  // moderation
  moderationLogs ModerationLog[]

  @@index([role])
  @@index([isActive])
  @@map("users")
}

model Booking {
  id             String         @id @default(uuid()) @db.Uuid
  bookingCode    String         @unique @map("booking_code") @db.VarChar(30)

  userId         String         @map("user_id") @db.Uuid
  venueId        String         @map("venue_id") @db.Uuid
  courtId        String         @map("court_id") @db.Uuid

  bookingDate    DateTime       @map("booking_date") @db.Date

  subtotalAmount Decimal        @map("subtotal_amount") @db.Decimal(12, 2)
  adminFeeAmount Decimal        @default(0) @map("admin_fee_amount") @db.Decimal(12, 2)
  totalAmount    Decimal        @map("total_amount") @db.Decimal(12, 2)

  bookingStatus  BookingStatus  @default(PENDING_PAYMENT) @map("booking_status")
  paymentStatus  PaymentStatus  @default(PENDING) @map("payment_status")

  expiresAt      DateTime?      @map("expires_at") @db.Timestamptz(6)
  notes          String?

  confirmedAt    DateTime?      @map("confirmed_at") @db.Timestamptz(6)
  completedAt    DateTime?      @map("completed_at") @db.Timestamptz(6)
  cancelledAt    DateTime?      @map("cancelled_at") @db.Timestamptz(6)
  cancelledBy    String?        @map("cancelled_by") @db.Uuid
  cancelReason   String?        @map("cancel_reason")

  createdAt      DateTime       @default(now()) @map("created_at") @db.Timestamptz(6)
  updatedAt      DateTime       @updatedAt @map("updated_at") @db.Timestamptz(6)
  deletedAt      DateTime?      @map("deleted_at") @db.Timestamptz(6)

  user            User   @relation("BookingUser", fields: [userId], references: [id], onDelete: Restrict)
  venue           Venue  @relation(fields: [venueId], references: [id], onDelete: Restrict)
  court           Court  @relation(fields: [courtId], references: [id], onDelete: Restrict)
  cancelledByUser User?  @relation("BookingCancelledBy", fields: [cancelledBy], references: [id], onDelete: SetNull)

  slots             BookingSlot[]
  payment           BookingPayment?
  availabilitySlots CourtAvailabilitySlot[] @relation("BookingAvailabilitySlots")

  @@index([userId, bookingDate])
  @@index([venueId, bookingDate])
  @@index([courtId, bookingDate])
  @@index([bookingStatus])
  @@index([paymentStatus])
  @@map("bookings")
}

model MatchParticipant {
  id       String   @id @default(uuid()) @db.Uuid
  matchId  String   @map("match_id") @db.Uuid
  userId   String   @map("user_id") @db.Uuid
  joinedAt DateTime @default(now()) @map("joined_at") @db.Timestamptz(6)

  match Match @relation(fields: [matchId], references: [id], onDelete: Cascade)
  user  User  @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([matchId, userId])
  @@map("match_participants")
}

//////////////////////////////////////////////////////
// REVIEWS
//////////////////////////////////////////////////////

model VenueReview {
  id         String    @id @default(uuid()) @db.Uuid
  venueId    String    @map("venue_id") @db.Uuid
  userId     String    @map("user_id") @db.Uuid
  rating     Int
  reviewText String?   @map("review_text")

  createdAt  DateTime  @default(now()) @map("created_at") @db.Timestamptz(6)
  updatedAt  DateTime  @updatedAt @map("updated_at") @db.Timestamptz(6)
  deletedAt  DateTime? @map("deleted_at") @db.Timestamptz(6)

  venue Venue @relation(fields: [venueId], references: [id], onDelete: Cascade)
  user  User  @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([venueId, userId])
  @@index([venueId])
  @@index([userId])
  @@map("venue_reviews")
}

//////////////////////////////////////////////////////
// NOTIFICATIONS
//////////////////////////////////////////////////////

model Notification {
  id         String                  @id @default(uuid()) @db.Uuid
  userId     String                  @map("user_id") @db.Uuid
  type       NotificationType
  entityType NotificationEntityType? @map("entity_type")
  entityId   String?                 @map("entity_id") @db.Uuid
  title      String                  @db.VarChar(150)
  message    String
  isRead     Boolean                 @default(false) @map("is_read")
  readAt     DateTime?               @map("read_at") @db.Timestamptz(6)
  createdAt  DateTime                @default(now()) @map("created_at") @db.Timestamptz(6)

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@index([userId, isRead])
  @@index([type])
  @@map("notifications")
}

//////////////////////////////////////////////////////
// MODERATION / AUDIT
//////////////////////////////////////////////////////

model ModerationLog {
  id         String               @id @default(uuid()) @db.Uuid
  adminId    String               @map("admin_id") @db.Uuid
  targetType ModerationTargetType @map("target_type")
  targetId   String               @map("target_id") @db.Uuid
  action     ModerationActionType
  reason     String?
  createdAt  DateTime             @default(now()) @map("created_at") @db.Timestamptz(6)

  admin User @relation(fields: [adminId], references: [id], onDelete: Restrict)

  @@index([adminId])
  @@index([targetType, targetId])
  @@map("moderation_logs")
}
